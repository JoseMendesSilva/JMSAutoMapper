    public abstract class MapperBase : IMapper
    {
        private readonly Dictionary<Type, PropertyInfo[]> _propertyCache = new();
        private readonly Dictionary<(Type Source, Type Target), Delegate> _compiledExpressions = new();

        #region Métodos Base
      
        public T? Map<T>(object? source) where T : class, new()
        {
            if (source == null) return null;
            return Map<T>(source);
        }

        private T? MapInternal<T>(object? source, HashSet<object> mappedObjects) where T : class, new()
        {
            if (source == null || mappedObjects.Contains(source))
                return null;

            mappedObjects.Add(source);
            return Map<T>(source);
        }

        #endregion

        #region Coleções Padrão

        public IEnumerable<T>? MapIEnumerable<T>(object? source) where T : class, new()
        {
            if (source == null) return null;
            if (source is not IEnumerable enumerable)
                throw new ArgumentException("O objeto de origem deve ser uma coleção.", nameof(source));

            var mappedObjects = new HashSet<object>();
            return enumerable.Cast<object>()
                .Select(item => MapInternal<T>(item, mappedObjects))
                .Where(result => result != null)
                .ToList()!;
        }

        public List<T>? MapList<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable?.ToList();
        }

        public ICollection<T>? MapICollection<T>(object? source) where T : class, new()
        {
            return MapList<T>(source);
        }

        public IReadOnlyList<T>? MapIReadOnlyList<T>(object? source) where T : class, new()
        {
            return MapList<T>(source);
        }

        public IReadOnlyCollection<T>? MapIReadOnlyCollection<T>(object? source) where T : class, new()
        {
            return MapList<T>(source);
        }

        public T[]? MapArray<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable?.ToArray();
        }

        public HashSet<T>? MapHashSet<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable?.ToHashSet();
        }

        #endregion

        #region Dicionários

        public Dictionary<TKey, TValue>? MapDictionary<TKey, TValue>(object? source)
            where TKey : notnull
            where TValue : class, new()
        {
            if (source == null) return null;
            if (source is not IDictionary dictionary)
                throw new ArgumentException("O objeto de origem deve ser um dicionário.", nameof(source));

            var result = new Dictionary<TKey, TValue>();
            var mappedObjects = new HashSet<object>();

            foreach (DictionaryEntry entry in dictionary)
            {
                // Tratamento especial para a chave
                TKey key;
                if (entry.Key is TKey typedKey)
                {
                    key = typedKey;
                }
                else
                {
                    try
                    {
                        key = (TKey)Convert.ChangeType(entry.Key, typeof(TKey));
                    }
                    catch
                    {
                        continue; // Ignora entradas com chaves inválidas
                    }
                }

                // Mapeamento do valor (que tem a restrição class)
                var value = MapInternal<TValue>(entry.Value, mappedObjects);
                if (value != null)
                {
                    result.Add(key, value);
                }
            }

            return result;
        }

        #endregion

        #region Coleções Thread-Safe

        public ConcurrentDictionary<TKey, TValue>? MapConcurrentDictionary<TKey, TValue>(object? source)
            where TKey : notnull
            where TValue : class, new()
        {
            var dictionary = MapDictionary<TKey, TValue>(source);
            return dictionary != null ? new ConcurrentDictionary<TKey, TValue>(dictionary) : null;
        }

        public BlockingCollection<T>? MapBlockingCollection<T>(object? source, int? boundedCapacity = null)
            where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            if (enumerable == null) return null;

            return boundedCapacity.HasValue
                ? new BlockingCollection<T>(new ConcurrentQueue<T>(enumerable), boundedCapacity.Value)
                : new BlockingCollection<T>(new ConcurrentQueue<T>(enumerable));
        }

        #endregion

        #region Coleções Imutáveis

        public ImmutableList<T>? MapImmutableList<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable?.ToImmutableList();
        }

        public ImmutableDictionary<TKey, TValue>? MapImmutableDictionary<TKey, TValue>(object? source)
            where TKey : notnull
            where TValue : class, new()
        {
            var dictionary = MapDictionary<TKey, TValue>(source);
            return dictionary?.ToImmutableDictionary();
        }

        public ImmutableArray<T>? MapImmutableArray<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable?.ToImmutableArray();
        }

        public ImmutableQueue<T>? MapImmutableQueue<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable != null ? ImmutableQueue.CreateRange(enumerable) : null;
        }

        public ImmutableStack<T>? MapImmutableStack<T>(object? source) where T : class, new()
        {
            var enumerable = MapIEnumerable<T>(source);
            return enumerable != null ? ImmutableStack.CreateRange(enumerable) : null;
        }

        #endregion

        #region Métodos Assíncronos

        public async Task<T?> MapAsync<T>(object? source) where T : class, new()
        {
            if (source == null) return null;
            return await Task.Run(() => Map<T>(source));
        }

        public async Task<IEnumerable<T>?> MapIEnumerableAsync<T>(object? source, int? maxDegreeOfParallelism = null)
            where T : class, new()
        {
            if (source == null) return null;
            if (source is not IEnumerable enumerable)
                throw new ArgumentException("O objeto de origem deve ser uma coleção.", nameof(source));

            var options = new ParallelOptions
            {
                MaxDegreeOfParallelism = maxDegreeOfParallelism ?? Environment.ProcessorCount
            };

            var results = new List<T>();
            var sourceList = enumerable.Cast<object>().ToList();

            await Parallel.ForEachAsync(sourceList, options, async (item, cancellationToken) =>
            {
                var mappedItem = await MapAsync<T>(item);
                if (mappedItem != null)
                {
                    lock (results)
                    {
                        results.Add(mappedItem);
                    }
                }
            });

            return results;
        }

        #endregion

        #region MemoryCache
        private readonly MemoryCache _cache = MemoryCache.Default;

        public CacheItem? MapToCacheItem<T>(object? source, string key, CacheItemPolicy? policy = null)
            where T : class, new()
        {
            if (source == null) return null;

            var mappedItem = Map<T>(source);
            if (mappedItem == null) return null;

            var cacheItem = new CacheItem(key, mappedItem);

            if (policy != null)
            {
                _cache.Set(cacheItem, policy);
            }

            return cacheItem;
        }

        #endregion

        protected abstract T? MapObject<T>(object source, HashSet<object> visited) where T : class, new();

        protected object? ConvertValue(object? value, Type targetType)
        {
            if (value == null)
            {
                if (Nullable.GetUnderlyingType(targetType) != null || !targetType.IsValueType)
                    return null;
                return Activator.CreateInstance(targetType);
            }

            try
            {
                var underlyingType = Nullable.GetUnderlyingType(targetType) ?? targetType;

                if (underlyingType.IsEnum && value is int intValue)
                    return Enum.ToObject(underlyingType, intValue);

                return Convert.ChangeType(value, underlyingType);
            }
            catch
            {
                return null;
            }
        }

        protected PropertyInfo[] GetProperties(Type type)
        {
            if (!_propertyCache.TryGetValue(type, out var properties))
            {
                properties = type.GetProperties(BindingFlags.Public | BindingFlags.Instance);
                _propertyCache[type] = properties;
            }
            return properties;
        }

        protected Delegate GetOrCreateCompiledExpression(Type sourceType, Type targetType, PropertyInfo sourceProperty, PropertyInfo targetProperty)
        {
            if (!_compiledExpressions.TryGetValue((sourceType, targetType), out var compiledExpression))
            {
                var sourceParam = Expression.Parameter(sourceType, "source");
                var sourcePropertyAccess = Expression.Property(sourceParam, sourceProperty);
                var convertExpression = Expression.Convert(sourcePropertyAccess, targetProperty.PropertyType);
                var lambda = Expression.Lambda(convertExpression, sourceParam);

                compiledExpression = lambda.Compile();
                _compiledExpressions[(sourceType, targetType)] = compiledExpression;
            }

            return compiledExpression;
        }
    }

    public class JMSMapper : MapperBase
    {
        private readonly Dictionary<(Type Source, Type Target), Dictionary<string, string>> _customMappings = new();
       
        private readonly Action<string, Exception>? _logger;

        public JMSMapper(Action<string, Exception>? logger = null)
        {
            _logger = logger;
        }

        public void AddCustomMapping<TSource, TTarget>(string sourcePropertyName, string targetPropertyName)
        {
            var key = (typeof(TSource), typeof(TTarget));
            if (!_customMappings.TryGetValue(key, out var mappings))
            {
                mappings = new Dictionary<string, string>();
                _customMappings[key] = mappings;
            }
            mappings[sourcePropertyName] = targetPropertyName;
        }

        protected override T? MapObject<T>(object source, HashSet<object> visited) where T : class
        {
            if (source == null || visited.Contains(source)) return null;

            visited.Add(source);

            T result = new();

            var sourceType = source.GetType();
            var targetType = typeof(T);

            var sourceProperties = GetProperties(sourceType);
            var targetProperties = GetProperties(targetType);

            foreach (var targetProperty in targetProperties)
            {
                if (!targetProperty.CanWrite) continue;

                var sourcePropertyName = GetMappedPropertyName(sourceType, targetType, targetProperty.Name);
                var sourceProperty = sourceProperties.FirstOrDefault(p => string.Equals(p.Name, sourcePropertyName, StringComparison.OrdinalIgnoreCase));

                if (sourceProperty == null || !sourceProperty.CanRead) continue;

                try
                {
                    var value = sourceProperty.GetValue(source);

                    if (value == null) continue;

                    if (IsCollection(targetProperty.PropertyType))
                    {
                        MapCollection(targetProperty, result, value, visited);
                    }
                    else if (IsComplexType(targetProperty.PropertyType))
                    {
                        MapComplexObject(targetProperty, result, value, visited);
                    }
                    else
                    {
                        var convertedValue = ConvertValue(value, targetProperty.PropertyType);
                        targetProperty.SetValue(result, convertedValue);
                    }
                }
                catch (Exception ex)
                {
                    Debug.WriteLine($"Erro ao mapear propriedade '{targetProperty.Name}': {ex.Message}");
                }
            }

            return result;
        }

        private string GetMappedPropertyName(Type sourceType, Type targetType, string targetPropertyName)
        {
            if (_customMappings.TryGetValue((sourceType, targetType), out var mappings) && mappings.TryGetValue(targetPropertyName, out var sourcePropertyName))
            {
                return sourcePropertyName;
            }
            return targetPropertyName;
        }

        private void MapCollection(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
        {
            var itemType = GetCollectionItemType(targetProperty.PropertyType);
            if (itemType == null || !(value is IEnumerable sourceCollection)) return;

            var mappedCollection = CreateCollection(targetProperty.PropertyType, itemType);
            if (mappedCollection == null) return;

            foreach (var item in sourceCollection)
            {
                if (item == null) continue;

                var mappedItem = MapObject(itemType, item, visited);
                if (mappedItem != null)
                    mappedCollection.Add(mappedItem);
            }

            targetProperty.SetValue(result, mappedCollection);
        }

        private object? MapObject(Type targetType, object source, HashSet<object> visited)
        {
            var method = typeof(JMSMapper)
                .GetMethod(nameof(MapObjectGeneric), BindingFlags.Instance | BindingFlags.NonPublic)
                ?.MakeGenericMethod(targetType);

            return method?.Invoke(this, new object[] { source, visited });
        }

        private T? MapObjectGeneric<T>(object source, HashSet<object> visited) where T : class, new()
        {
            return MapObject<T>(source, visited);
        }

        private Type? GetCollectionItemType(Type collectionType)
        {
            if (collectionType.IsArray)
                return collectionType.GetElementType();

            if (collectionType.IsGenericType)
                return collectionType.GetGenericArguments().FirstOrDefault();

            return null;
        }

        private IList? CreateCollection(Type collectionType, Type itemType)
        {
            if (collectionType.IsArray)
                return Array.CreateInstance(itemType, 0) as IList;

            if (collectionType.IsGenericType)
                return Activator.CreateInstance(typeof(List<>).MakeGenericType(itemType)) as IList;

            return null;
        }

        private void MapComplexObject(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
        {
            var mappedObject = MapObject(targetProperty.PropertyType, value, visited);
            targetProperty.SetValue(result, mappedObject);
        }

        private bool IsCollection(Type type)
        {
            return type != typeof(string) && typeof(IEnumerable).IsAssignableFrom(type);
        }

        private bool IsComplexType(Type type)
        {
            return !type.IsPrimitive && !type.IsValueType && type != typeof(string);
        }
    }

Análise do Código JMSAutoMapper
Visão Geral
O código implementa um sistema de mapeamento de objetos (object mapping) em C#, similar ao AutoMapper, mas com funcionalidades adicionais. Ele permite converter objetos de um tipo para outro, incluindo suporte para coleções complexas, dicionários, coleções thread-safe e imutáveis.

Principais Componentes
1. Interface IMapper
Define a interface pública com todos os métodos de mapeamento disponíveis, incluindo:

Mapeamento básico de objetos (Map<T>)

Mapeamento de coleções (listas, arrays, conjuntos)

Mapeamento de dicionários

Coleções thread-safe (ConcurrentDictionary, BlockingCollection)

Coleções imutáveis (ImmutableList, ImmutableDictionary, etc.)

Métodos assíncronos

Integração com MemoryCache

2. Classe Abstrata MapperBase
Implementa a maior parte da funcionalidade definida em IMapper, fornecendo:

Cache de propriedades e expressões compiladas para melhor performance

Implementações base para todos os métodos de mapeamento

Métodos auxiliares para conversão de valores e manipulação de propriedades

3. Classe JMSMapper
Implementação concreta que herda de MapperBase, adicionando:

Suporte a mapeamentos personalizados entre propriedades

Lógica de mapeamento de objetos complexos

Tratamento de coleções e tipos complexos

Possibilidade de logging através de um delegate

Pontos Fortes
Extensibilidade: Suporta uma grande variedade de tipos de coleções, desde as básicas até as mais especializadas (thread-safe, imutáveis).

Performance: Utiliza cache de propriedades e expressões compiladas para melhorar a performance do mapeamento.

Segurança: Implementa tratamento de referências circulares através do HashSet<object> visited.

Flexibilidade: Permite mapeamentos personalizados entre propriedades com nomes diferentes.

Suporte Assíncrono: Oferece versões assíncronas para operações de mapeamento.

Integração com Cache: Possui métodos específicos para integração com MemoryCache.

Possíveis Melhorias
Injeção de Dependência: Poderia ser mais fácil de integrar com sistemas de DI.

Documentação XML: Falta documentação XML nos métodos para facilitar o uso.

Validação mais Rigorosa: Algumas validações de tipo poderiam ser mais precisas.

Suporte a Fluent API: Para configuração de mapeamentos, similar ao AutoMapper.

Profile de Mapeamento: Possibilidade de organizar mapeamentos em "profiles" separados.

Testes Unitários: Seria importante ter testes abrangentes para garantir o funcionamento correto.

Casos de Uso
Este mapeador seria útil em cenários como:

Conversão entre modelos de domínio e DTOs

Integração entre sistemas com modelos de dados diferentes

Transformação de dados para APIs

Caching de objetos mapeados

Processamento paralelo de coleções de dados

Conclusão
O JMSAutoMapper é uma implementação robusta e abrangente de um mapeador de objetos, com funcionalidades que vão além das soluções mais básicas. Sua arquitetura bem organizada e extensível permite fácil adaptação para necessidades específicas, enquanto sua variedade de métodos de mapeamento cobre a maioria dos cenários comuns de transformação de dados em aplicações .NET.

    /// <summary>
    /// Implementação base do mapeador
    /// </summary>
    public abstract class MapperBase : IMapper
    {
        private readonly Dictionary<Type, PropertyInfo[]> _propertyCache = new();
        private readonly Dictionary<(Type Source, Type Target), Delegate> _compiledExpressions = new();
        private readonly int _maxRecursionDepth = 50; // Prevenção contra StackOverflow

        #region Métodos Base
      
        /// <inheritdoc/>
        public T? Map<T>(object? source) where T : class, new()
        {
            if (source == null) return null;
            
            var mappedObjects = new Dictionary<object, object>();
            return MapInternal<T>(source, mappedObjects, 0);
        }

        private T? MapInternal<T>(object? source, IDictionary<object, object> mappedObjects, int currentDepth) 
            where T : class, new()
        {
            if (source == null) return null;
            
            // Prevenção contra StackOverflow
            if (currentDepth > _maxRecursionDepth)
            {
                throw new InvalidOperationException(
                    $"Profundidade de recursão excedida. Limite: {_maxRecursionDepth}. " +
                    "Possível referência circular não tratada.");
            }

            // Verifica se o objeto já foi mapeado
            if (mappedObjects.TryGetValue(source, out var mappedObject))
            {
                return mappedObject as T;
            }

            var result = MapObject<T>(source, mappedObjects, currentDepth);
            if (result != null)
            {
                mappedObjects[source] = result;
            }

            return result;
        }

        #endregion

        #region Coleções Padrão

        /// <inheritdoc/>
        public IEnumerable<T>? MapIEnumerable<T>(object? source) where T : class, new()
        {
            if (source == null) return null;
            if (source is not IEnumerable enumerable)
                throw new ArgumentException("O objeto de origem deve ser uma coleção.", nameof(source));

            var mappedObjects = new Dictionary<object, object>();
            return enumerable.Cast<object>()
                .Select(item => MapInternal<T>(item, mappedObjects, 0))
                .Where(result => result != null)
                .ToList()!;
        }

        // Implementações dos outros métodos de coleção permanecem iguais, mas usando MapInternal com currentDepth

        #endregion

        // Restante das implementações permanecem similares, mas atualizadas para usar o novo MapInternal

        protected abstract T? MapObject<T>(object source, IDictionary<object, object> mappedObjects, int currentDepth) 
            where T : class, new();

        // Métodos auxiliares permanecem os mesmos
    }

    /// <summary>
    /// Implementação concreta do mapeador com suporte a Fluent API e Profiles
    /// </summary>
    public class JMSMapper : MapperBase, IMapper, IMapperConfigurationExpression
    {
        private readonly Dictionary<(Type Source, Type Target), Dictionary<string, string>> _propertyMappings = new();
        private readonly Dictionary<(Type Source, Type Target), Dictionary<string, Func<object, object>>> _customMappings = new();
        private readonly List<Profile> _profiles = new();
        private readonly Action<string, Exception>? _logger;

        public JMSMapper(Action<string, Exception>? logger = null)
        {
            _logger = logger;
        }

        /// <summary>
        /// Configura o mapeador com os profiles registrados
        /// </summary>
        public void Initialize()
        {
            foreach (var profile in _profiles)
            {
                profile.Configure();
            }
        }

        #region Implementação IMapperConfigurationExpression

        public IMappingExpression<TSource, TDestination> CreateMap<TSource, TDestination>() 
            where TDestination : class, new()
        {
            return new MappingExpression<TSource, TDestination>(this);
        }

        public void AddProfile(Profile profile)
        {
            if (profile == null) throw new ArgumentNullException(nameof(profile));
            _profiles.Add(profile);
            profile.Configure();
        }

        #endregion

        #region Implementação dos métodos de mapeamento

        protected override T? MapObject<T>(object source, IDictionary<object, object> mappedObjects, int currentDepth) 
            where T : class, new()
        {
            if (source == null || mappedObjects.ContainsKey(source)) return null;

            T result = new();

            var sourceType = source.GetType();
            var targetType = typeof(T);

            var sourceProperties = GetProperties(sourceType);
            var targetProperties = GetProperties(targetType);

            foreach (var targetProperty in targetProperties)
            {
                if (!targetProperty.CanWrite) continue;

                try
                {
                    // Verifica mapeamentos customizados primeiro
                    if (TryGetCustomMapping(source, sourceType, targetType, targetProperty, out var value))
                    {
                        targetProperty.SetValue(result, value);
                        continue;
                    }

                    // Mapeamento padrão por nome
                    var sourcePropertyName = GetMappedPropertyName(sourceType, targetType, targetProperty.Name);
                    var sourceProperty = sourceProperties.FirstOrDefault(p => 
                        string.Equals(p.Name, sourcePropertyName, StringComparison.OrdinalIgnoreCase));

                    if (sourceProperty == null || !sourceProperty.CanRead) continue;

                    var sourceValue = sourceProperty.GetValue(source);
                    if (sourceValue == null) continue;

                    if (IsCollection(targetProperty.PropertyType))
                    {
                        MapCollection(targetProperty, result, sourceValue, mappedObjects, currentDepth);
                    }
                    else if (IsComplexType(targetProperty.PropertyType))
                    {
                        MapComplexObject(targetProperty, result, sourceValue, mappedObjects, currentDepth);
                    }
                    else
                    {
                        var convertedValue = ConvertValue(sourceValue, targetProperty.PropertyType);
                        targetProperty.SetValue(result, convertedValue);
                    }
                }
                catch (Exception ex)
                {
                    _logger?.Invoke($"Erro ao mapear propriedade '{targetProperty.Name}'", ex);
                }
            }

            return result;
        }

        private bool TryGetCustomMapping(
            object source, Type sourceType, Type targetType, PropertyInfo targetProperty, out object? value)
        {
            value = null;
            
            // Verifica mapeamentos por função
            if (_customMappings.TryGetValue((sourceType, targetType), out var funcMappings) &&
                funcMappings.TryGetValue(targetProperty.Name, out var mappingFunc))
            {
                value = mappingFunc(source);
                return true;
            }

            return false;
        }

        // Implementações dos outros métodos auxiliares atualizadas para usar mappedObjects e currentDepth

        #endregion

        private class MappingExpression<TSource, TDestination> : IMappingExpression<TSource, TDestination>
            where TDestination : class, new()
        {
            private readonly JMSMapper _mapper;

            public MappingExpression(JMSMapper mapper)
            {
                _mapper = mapper;
            }

            public IMappingExpression<TSource, TDestination> ForMember<TMember>(
                Expression<Func<TDestination, TMember>> destinationMember,
                Action<IMemberConfigurationExpression<TSource, TDestination, TMember>> memberOptions)
            {
                var memberName = GetMemberName(destinationMember);
                var expression = new MemberConfigurationExpression<TSource, TDestination, TMember>(memberName);
                memberOptions(expression);

                expression.Configure(_mapper);
                return this;
            }

            private static string GetMemberName(Expression<Func<TDestination, TMember>> expression)
            {
                if (expression.Body is MemberExpression memberExpression)
                {
                    return memberExpression.Member.Name;
                }
                throw new ArgumentException("Expression must be a member access expression", nameof(expression));
            }
        }

        private class MemberConfigurationExpression<TSource, TDestination, TMember> 
            : IMemberConfigurationExpression<TSource, TDestination, TMember>
            where TDestination : class, new()
        {
            private readonly string _memberName;
            private string? _sourcePropertyName;
            private Func<TSource, TMember>? _mappingFunction;

            public MemberConfigurationExpression(string memberName)
            {
                _memberName = memberName;
            }

            public void MapFrom(string sourcePropertyName)
            {
                _sourcePropertyName = sourcePropertyName;
            }

            public void MapFrom(Func<TSource, TMember> mappingFunction)
            {
                _mappingFunction = mappingFunction;
            }

            public void Configure(JMSMapper mapper)
            {
                var sourceType = typeof(TSource);
                var targetType = typeof(TDestination);

                if (_sourcePropertyName != null)
                {
                    // Mapeamento por nome de propriedade
                    if (!mapper._propertyMappings.TryGetValue((sourceType, targetType), out var mappings))
                    {
                        mappings = new Dictionary<string, string>();
                        mapper._propertyMappings[(sourceType, targetType)] = mappings;
                    }
                    mappings[_memberName] = _sourcePropertyName;
                }
                else if (_mappingFunction != null)
                {
                    // Mapeamento por função customizada
                    if (!mapper._customMappings.TryGetValue((sourceType, targetType), out var funcMappings))
                    {
                        funcMappings = new Dictionary<string, Func<object, object>>();
                        mapper._customMappings[(sourceType, targetType)] = funcMappings;
                    }
                    funcMappings[_memberName] = src => _mappingFunction((TSource)src)!;
                }
            }
        }
    }

    protected object? ConvertValue(object? value, Type targetType)
    {
        if (value == null)
        {
            if (Nullable.GetUnderlyingType(targetType) != null || !targetType.IsValueType)
                return null;
            return Activator.CreateInstance(targetType);
        }

        try
        {
            var underlyingType = Nullable.GetUnderlyingType(targetType) ?? targetType;

            // Se o tipo de destino for string, converte o valor para string
            if (underlyingType == typeof(string))
            {
                return value.ToString();
            }

            // Se o valor já for do tipo correto, retorna o valor
            if (underlyingType.IsInstanceOfType(value))
            {
                return value;
            }

            // Conversão para enums
            if (underlyingType.IsEnum && value is int intValue)
            {
                return Enum.ToObject(underlyingType, intValue);
            }

            // Conversão para outros tipos
            return Convert.ChangeType(value, underlyingType);
        }
        catch
        {
            return null;
        }
    }

public class JMSMapper : MapperBase
{
    private readonly Dictionary<(Type Source, Type Target), Dictionary<string, string>> _customMappings = new();
    private readonly Action<string, Exception>? _logger;

    public JMSMapper(Action<string, Exception>? logger = null)
    {
        _logger = logger;
    }

    public void AddCustomMapping<TSource, TTarget>(string sourcePropertyName, string targetPropertyName)
    {
        var key = (typeof(TSource), typeof(TTarget));
        if (!_customMappings.TryGetValue(key, out var mappings))
        {
            mappings = new Dictionary<string, string>();
            _customMappings[key] = mappings;
        }
        mappings[sourcePropertyName] = targetPropertyName;
    }

    protected override T? MapObject<T>(object source, HashSet<object> visited) where T : class
    {
        if (source == null || visited.Contains(source)) return null;

        visited.Add(source);

        T result = new();

        var sourceType = source.GetType();
        var targetType = typeof(T);

        var sourceProperties = GetProperties(sourceType);
        var targetProperties = GetProperties(targetType);

        foreach (var targetProperty in targetProperties)
        {
            if (!targetProperty.CanWrite) continue;

            var sourcePropertyName = GetMappedPropertyName(sourceType, targetType, targetProperty.Name);
            var sourceProperty = sourceProperties.FirstOrDefault(p => string.Equals(p.Name, sourcePropertyName, StringComparison.OrdinalIgnoreCase));

            if (sourceProperty == null || !sourceProperty.CanRead) continue;

            try
            {
                var value = sourceProperty.GetValue(source);

                if (value == null) continue;

                if (IsCollection(targetProperty.PropertyType))
                {
                    MapCollection(targetProperty, result, value, visited);
                }
                else if (IsComplexType(targetProperty.PropertyType))
                {
                    MapComplexObject(targetProperty, result, value, visited);
                }
                else
                {
                    var convertedValue = ConvertValue(value, targetProperty.PropertyType);
                    targetProperty.SetValue(result, convertedValue);
                }
            }
            catch (Exception ex)
            {
                _logger?.Invoke($"Erro ao mapear propriedade '{targetProperty.Name}'", ex);
            }
        }

        return result;
    }

    private string GetMappedPropertyName(Type sourceType, Type targetType, string targetPropertyName)
    {
        if (_customMappings.TryGetValue((sourceType, targetType), out var mappings) && mappings.TryGetValue(targetPropertyName, out var sourcePropertyName))
        {
            return sourcePropertyName;
        }
        return targetPropertyName;
    }

    private void MapCollection(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var itemType = GetCollectionItemType(targetProperty.PropertyType);
        if (itemType == null || !(value is IEnumerable sourceCollection)) return;

        var mappedCollection = CreateCollection(targetProperty.PropertyType, itemType);
        if (mappedCollection == null) return;

        foreach (var item in sourceCollection)
        {
            if (item == null) continue;

            var mappedItem = MapObject(itemType, item, visited);
            if (mappedItem != null)
                mappedCollection.Add(mappedItem);
        }

        targetProperty.SetValue(result, mappedCollection);
    }

    private object? MapObject(Type targetType, object source, HashSet<object> visited)
    {
        var method = typeof(JMSMapper)
            .GetMethod(nameof(MapObjectGeneric), BindingFlags.Instance | BindingFlags.NonPublic)
            ?.MakeGenericMethod(targetType);

        return method?.Invoke(this, new object[] { source, visited });
    }

    private T? MapObjectGeneric<T>(object source, HashSet<object> visited) where T : class, new()
    {
        return MapObject<T>(source, visited);
    }

    private Type? GetCollectionItemType(Type collectionType)
    {
        if (collectionType.IsArray)
            return collectionType.GetElementType();

        if (collectionType.IsGenericType)
            return collectionType.GetGenericArguments().FirstOrDefault();

        return null;
    }

    private IList? CreateCollection(Type collectionType, Type itemType)
    {
        if (collectionType.IsArray)
            return Array.CreateInstance(itemType, 0) as IList;

        if (collectionType.IsGenericType)
            return Activator.CreateInstance(typeof(List<>).MakeGenericType(itemType)) as IList;

        return null;
    }

    private void MapComplexObject(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var mappedObject = MapObject(targetProperty.PropertyType, value, visited);
        targetProperty.SetValue(result, mappedObject);
    }

    private bool IsCollection(Type type)
    {
        return type != typeof(string) && typeof(IEnumerable).IsAssignableFrom(type);
    }

    private bool IsComplexType(Type type) //
    {
        return !type.IsPrimitive && !type.IsValueType && type != typeof(string);
    }

    ImmutableDictionary<TKey, TValue>? MapImmutableDictionary<TKey, TValue>(object? source)
        where TKey : notnull
        where TValue : class, new();
    ImmutableArray<T>? MapImmutableArray<T>(object? source) where T : class, new();
    ImmutableQueue<T>? MapImmutableQueue<T>(object? source) where T : class, new();
    ImmutableStack<T>? MapImmutableStack<T>(object? source) where T : class, new();

    // Métodos assíncronos
    Task<T?> MapAsync<T>(object? source) where T : class, new();
    Task<IEnumerable<T>?> MapIEnumerableAsync<T>(object? source, int? maxDegreeOfParallelism = null)
        where T : class, new();

    // Configuração de mapeamento
    IMappingExpression<TSource, TDestination> CreateMap<TSource, TDestination>() 
        where TSource : class 
        where TDestination : class, new();

public static class MapperExtensions
{
    public static IServiceCollection AddJMSMapper(this IServiceCollection services, Action<MapperConfiguration>? configure = null)
    {
        var config = new MapperConfiguration();
        configure?.Invoke(config);
        
        services.AddSingleton<IMapper>(provider => 
        {
            var logger = provider.GetService<Action<string, Exception>>();
            return new JMSMapper(config, logger);
        });
        
        return services;
    }

    public static void Initialize(Action<MapperConfiguration> configure)
    {
        var config = new MapperConfiguration();
        configure.Invoke(config);
        // Configuração é armazenada estáticamente no JMSMapper
        JMSMapper.SetConfiguration(config);
    }
}

public class MapperConfiguration
{
    internal Dictionary<(Type Source, Type Target), Dictionary<string, Func<object, object>>> CustomMappings { get; } = new();
    internal Dictionary<(Type Source, Type Target), Dictionary<string, string>> PropertyMappings { get; } = new();

    public IMappingExpression<TSource, TDestination> CreateMap<TSource, TDestination>() 
        where TSource : class 
        where TDestination : class, new()
    {
        return new MappingExpression<TSource, TDestination>(this);
    }
}

public class MappingExpression<TSource, TDestination> : IMappingExpression<TSource, TDestination> 
    where TSource : class 
    where TDestination : class, new()
{
    private readonly MapperConfiguration _config;

    public MappingExpression(MapperConfiguration config)
    {
        _config = config;
    }

    public IMappingExpression<TSource, TDestination> ForMember<TMember>(
        string destinationProperty,
        Func<TSource, TMember> mappingFunction)
    {
        var key = (typeof(TSource), typeof(TDestination));
        
        if (!_config.CustomMappings.TryGetValue(key, out var mappings))
        {
            mappings = new Dictionary<string, Func<object, object>>();
            _config.CustomMappings[key] = mappings;
        }
        
        mappings[destinationProperty] = src => mappingFunction((TSource)src)!;
        
        return this;
    }

    public IMappingExpression<TSource, TDestination> ForMember(
        string destinationProperty,
        string sourceProperty)
    {
        var key = (typeof(TSource), typeof(TDestination));
        
        if (!_config.PropertyMappings.TryGetValue(key, out var mappings))
        {
            mappings = new Dictionary<string, string>();
            _config.PropertyMappings[key] = mappings;
        }
        
        mappings[destinationProperty] = sourceProperty;
        
        return this;
    }
}

public abstract class MapperBase : IMapper
{
    private readonly Dictionary<Type, PropertyInfo[]> _propertyCache = new();
    private readonly Dictionary<(Type Source, Type Target), Delegate> _compiledExpressions = new();

    public T? Map<T>(object? source) where T : class, new()
    {
        if (source == null) return null;
        return MapObject<T>(source, new HashSet<object>());
    }

    protected abstract T? MapObject<T>(object source, HashSet<object> visited) where T : class, new();

    protected object? ConvertValue(object? value, Type targetType)
    {
        if (value == null)
        {
            if (Nullable.GetUnderlyingType(targetType) != null || !targetType.IsValueType)
                return null;
            return Activator.CreateInstance(targetType);
        }

        try
        {
            var underlyingType = Nullable.GetUnderlyingType(targetType) ?? targetType;

            if (underlyingType == typeof(string))
            {
                return value.ToString();
            }

            if (underlyingType.IsInstanceOfType(value))
            {
                return value;
            }

            if (underlyingType.IsEnum && value is int intValue)
            {
                return Enum.ToObject(underlyingType, intValue);
            }

            return Convert.ChangeType(value, underlyingType);
        }
        catch
        {
            return null;
        }
    }

    protected PropertyInfo[] GetProperties(Type type)
    {
        if (!_propertyCache.TryGetValue(type, out var properties))
        {
            properties = type.GetProperties(BindingFlags.Public | BindingFlags.Instance);
            _propertyCache[type] = properties;
        }
        return properties;
    }

    public IEnumerable<TDestination>? MapIEnumerable<TDestination>(object? source) where TDestination : class, new()
    {
        if (source == null) return null;
        if (source is not IEnumerable enumerable)
            throw new ArgumentException("Source must be a collection", nameof(source));

        return enumerable.Cast<object>()
            .Select(item => Map<TDestination>(item))
            .Where(result => result != null)
            .ToList()!;
    }

    public List<T>? MapList<T>(object? source) where T : class, new()
    {
        var list = new List<T>();

        if (source == null) return null;
        foreach (var item in (IList)source)
        {
            var result = Map<T>(item);
            list.Add(result!);
        }

        return list;
    }

    public ICollection<T>? MapICollection<T>(object? source) where T : class, new()
    {
        return MapList<T>(source);
    }

    public IReadOnlyList<T>? MapIReadOnlyList<T>(object? source) where T : class, new()
    {
        return MapList<T>(source);
    }

    public IReadOnlyCollection<T>? MapIReadOnlyCollection<T>(object? source) where T : class, new()
    {
        return MapList<T>(source);
    }

    public T[]? MapArray<T>(object? source) where T : class, new()
    {
        var list = MapList<T>(source);
        return list?.ToArray();
    }

    public HashSet<T>? MapHashSet<T>(object? source) where T : class, new()
    {
        var list = MapList<T>(source);
        return list != null ? new HashSet<T>(list) : null;
    }

    public Dictionary<TKey, TValue>? MapDictionary<TKey, TValue>(object? source)
        where TKey : notnull
        where TValue : class, new()
    {
        if (source == null) return null;
        if (source is not IDictionary dictionary)
            throw new ArgumentException("Source must be a dictionary", nameof(source));

        var result = new Dictionary<TKey, TValue>();
        foreach (DictionaryEntry entry in dictionary)
        {
            if (entry.Key is TKey key && entry.Value != null)
            {
                var value = Map<TValue>(entry.Value);
                if (value != null)
                    result.Add(key, value);
            }
        }
        return result;
    }

    public ImmutableList<T>? MapImmutableList<T>(object? source) where T : class, new()
    {
        var list = MapList<T>(source);
        return list?.ToImmutableList();
    }

    public ImmutableDictionary<TKey, TValue>? MapImmutableDictionary<TKey, TValue>(object? source)
        where TKey : notnull
        where TValue : class, new()
    {
        var dict = MapDictionary<TKey, TValue>(source);
        return dict?.ToImmutableDictionary();
    }

    public ImmutableArray<T>? MapImmutableArray<T>(object? source) where T : class, new()
    {
        var list = MapList<T>(source);
        return list?.ToImmutableArray();
    }

    public ImmutableQueue<T>? MapImmutableQueue<T>(object? source) where T : class, new()
    {
        var list = MapList<T>(source);
        return list != null ? ImmutableQueue.CreateRange(list) : null;
    }

    public ImmutableStack<T>? MapImmutableStack<T>(object? source) where T : class, new()
    {
        var list = MapList<T>(source);
        return list != null ? ImmutableStack.CreateRange(list) : null;
    }

    public Task<T?> MapAsync<T>(object? source) where T : class, new()
    {
        return Task.FromResult(Map<T>(source));
    }

    public Task<IEnumerable<T>?> MapIEnumerableAsync<T>(object? source, int? maxDegreeOfParallelism = null) where T : class, new()
    {
        return Task.FromResult(MapIEnumerable<T>(source));
    }

    public IMappingExpression<TSource, TDestination> CreateMap<TSource, TDestination>() 
        where TSource : class 
        where TDestination : class, new()
    {
        throw new NotImplementedException("CreateMap should be called through MapperConfiguration");
    }
}

public class JMSMapper : MapperBase
{
    private static MapperConfiguration? _staticConfig;
    private readonly MapperConfiguration? _instanceConfig;
    private readonly Action<string, Exception>? _logger;

    public JMSMapper(MapperConfiguration? config = null, Action<string, Exception>? logger = null)
    {
        _instanceConfig = config;
        _logger = logger;
    }

    public static void SetConfiguration(MapperConfiguration config)
    {
        _staticConfig = config;
    }

    private MapperConfiguration? GetActiveConfig() => _instanceConfig ?? _staticConfig;

    protected override T? MapObject<T>(object source, HashSet<object> visited) where T : class, new()
    {
        if (source == null || visited.Contains(source)) return null;

        visited.Add(source);

        T result = new();

        var sourceType = source.GetType();
        var targetType = typeof(T);

        var sourceProperties = GetProperties(sourceType);
        var targetProperties = GetProperties(targetType);

        var config = GetActiveConfig();
        var customMappings = config?.CustomMappings;
        var propertyMappings = config?.PropertyMappings;

        foreach (var targetProperty in targetProperties)
        {
            if (!targetProperty.CanWrite) continue;

            try
            {
                // Verifica mapeamentos customizados primeiro
                var key = (sourceType, targetType);
                if (customMappings?.TryGetValue(key, out var mappings) == true && 
                    mappings.TryGetValue(targetProperty.Name, out var mappingFunc))
                {
                    var value = mappingFunc(source);
                    targetProperty.SetValue(result, ConvertValue(value, targetProperty.PropertyType));
                    continue;
                }

                // Verifica mapeamentos de propriedade
                var sourcePropertyName = GetMappedPropertyName(sourceType, targetType, targetProperty.Name, propertyMappings);
                var sourceProperty = sourceProperties.FirstOrDefault(p => 
                    string.Equals(p.Name, sourcePropertyName, StringComparison.OrdinalIgnoreCase));

                if (sourceProperty == null || !sourceProperty.CanRead) continue;

                var value = sourceProperty.GetValue(source);

                if (value == null) continue;

                if (IsCollection(targetProperty.PropertyType))
                {
                    MapCollection(targetProperty, result, value, visited);
                }
                else if (IsComplexType(targetProperty.PropertyType))
                {
                    MapComplexObject(targetProperty, result, value, visited);
                }
                else
                {
                    var convertedValue = ConvertValue(value, targetProperty.PropertyType);
                    targetProperty.SetValue(result, convertedValue);
                }
            }
            catch (Exception ex)
            {
                _logger?.Invoke($"Erro ao mapear propriedade '{targetProperty.Name}'", ex);
            }
        }

        return result;
    }

    private string GetMappedPropertyName(Type sourceType, Type targetType, string targetPropertyName, 
        Dictionary<(Type, Type), Dictionary<string, string>>? propertyMappings)
    {
        var key = (sourceType, targetType);
        if (propertyMappings?.TryGetValue(key, out var mappings) == true && 
            mappings.TryGetValue(targetPropertyName, out var sourcePropertyName))
        {
            return sourcePropertyName;
        }
        return targetPropertyName;
    }

    private void MapCollection(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var itemType = GetCollectionItemType(targetProperty.PropertyType);
        if (itemType == null || !(value is IEnumerable sourceCollection)) return;

        var mappedCollection = CreateCollection(targetProperty.PropertyType, itemType);
        if (mappedCollection == null) return;

        foreach (var item in sourceCollection)
        {
            if (item == null) continue;

            var mappedItem = MapObject(itemType, item, visited);
            if (mappedItem != null)
                mappedCollection.Add(mappedItem);
        }

        targetProperty.SetValue(result, mappedCollection);
    }

    private object? MapObject(Type targetType, object source, HashSet<object> visited)
    {
        var method = typeof(JMSMapper)
            .GetMethod(nameof(MapObjectGeneric), BindingFlags.Instance | BindingFlags.NonPublic)
            ?.MakeGenericMethod(targetType);

        return method?.Invoke(this, new object[] { source, visited });
    }

    private T? MapObjectGeneric<T>(object source, HashSet<object> visited) where T : class, new()
    {
        return MapObject<T>(source, visited);
    }

    private Type? GetCollectionItemType(Type collectionType)
    {
        if (collectionType.IsArray)
            return collectionType.GetElementType();

        if (collectionType.IsGenericType)
            return collectionType.GetGenericArguments().FirstOrDefault();

        return null;
    }

    private IList? CreateCollection(Type collectionType, Type itemType)
    {
        if (collectionType.IsArray)
            return Array.CreateInstance(itemType, 0) as IList;

        if (collectionType.IsGenericType)
        {
            var listType = typeof(List<>).MakeGenericType(itemType);
            return Activator.CreateInstance(listType) as IList;
        }

        return null;
    }

    private void MapComplexObject(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var mappedObject = MapObject(targetProperty.PropertyType, value, visited);
        targetProperty.SetValue(result, mappedObject);
    }

    private bool IsCollection(Type type)
    {
        return type != typeof(string) && typeof(IEnumerable).IsAssignableFrom(type);
    }

    private bool IsComplexType(Type type)
    {
        return !type.IsPrimitive && !type.IsValueType && type != typeof(string);
    }
}

Permite tanto mapeamento de propriedades quanto funções customizadas

csharp

// Mapeamento
var source = new SourceClass { FirstName = "John", LastName = "Doe", Years = 30 };
var destination = mapper.Map<DestinationClass>(source);

Configuração:

As chamadas CreateMap e ForMember registram os mapeamentos em dicionários internos

Os mapeamentos são armazenados em:

PropertyMappings: para mapeamentos simples de propriedade para propriedade

CustomMappings: para mapeamentos com funções customizadas

Mapeamento:

Quando Map é chamado, o mapeador:

Cria uma nova instância do tipo destino

Para cada propriedade no destino:

Verifica se há um mapeamento customizado

Se não, procura por mapeamento de propriedade

Se não, tenta mapear por nome convencional

Aplica conversões de tipo quando necessário

Trata corretamente coleções e tipos complexos aninhados

Injeção de Dependência:

O mapeador é registrado como singleton no container DI

Pode ser injetado em qualquer classe que necessite de mapeamento

A configuração é passada durante o registro
