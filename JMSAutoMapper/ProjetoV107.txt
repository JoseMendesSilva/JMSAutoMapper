        try

            // Se o tipo de destino for string, converte o valor para string

                return value.ToString();

                return value;

            {
                return Enum.ToObject(underlyingType, intValue);
            }

            // Conversão para outros tipos

        catch
        {
            return null;
        }

public class JMSMapper : MapperBase
{
    private readonly Dictionary<(Type Source, Type Target), Dictionary<string, string>> _customMappings = new();
    private readonly Action<string, Exception>? _logger;

    public JMSMapper(Action<string, Exception>? logger = null)
    {
        _logger = logger;
    }

    public void AddCustomMapping<TSource, TTarget>(string sourcePropertyName, string targetPropertyName)
    {
        var key = (typeof(TSource), typeof(TTarget));
        if (!_customMappings.TryGetValue(key, out var mappings))
        {
            mappings = new Dictionary<string, string>();
            _customMappings[key] = mappings;
        }
        mappings[sourcePropertyName] = targetPropertyName;
    }

    protected override T? MapObject<T>(object source, HashSet<object> visited) where T : class
    {
        if (source == null || visited.Contains(source)) return null;

        visited.Add(source);

        T result = new();

        var sourceType = source.GetType();
        var targetType = typeof(T);

        var sourceProperties = GetProperties(sourceType);
        var targetProperties = GetProperties(targetType);

        foreach (var targetProperty in targetProperties)
        {
            if (!targetProperty.CanWrite) continue;

            var sourcePropertyName = GetMappedPropertyName(sourceType, targetType, targetProperty.Name);
            var sourceProperty = sourceProperties.FirstOrDefault(p => string.Equals(p.Name, sourcePropertyName, StringComparison.OrdinalIgnoreCase));

            if (sourceProperty == null || !sourceProperty.CanRead) continue;

            try
            {
                var value = sourceProperty.GetValue(source);

                if (value == null) continue;

                if (IsCollection(targetProperty.PropertyType))
                {
                    MapCollection(targetProperty, result, value, visited);
                }
                else if (IsComplexType(targetProperty.PropertyType))
                {
                    MapComplexObject(targetProperty, result, value, visited);
                }
                else
                {
                    var convertedValue = ConvertValue(value, targetProperty.PropertyType);
                    targetProperty.SetValue(result, convertedValue);
                }
            }
            catch (Exception ex)
            {
                _logger?.Invoke($"Erro ao mapear propriedade '{targetProperty.Name}'", ex);
            }
        }

        return result;
    }

    private string GetMappedPropertyName(Type sourceType, Type targetType, string targetPropertyName)
    {
        if (_customMappings.TryGetValue((sourceType, targetType), out var mappings) && mappings.TryGetValue(targetPropertyName, out var sourcePropertyName))
        {
            return sourcePropertyName;
        }
        return targetPropertyName;
    }

    private void MapCollection(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var itemType = GetCollectionItemType(targetProperty.PropertyType);
        if (itemType == null || !(value is IEnumerable sourceCollection)) return;

        var mappedCollection = CreateCollection(targetProperty.PropertyType, itemType);
        if (mappedCollection == null) return;

        foreach (var item in sourceCollection)
        {
            if (item == null) continue;

            var mappedItem = MapObject(itemType, item, visited);
            if (mappedItem != null)
                mappedCollection.Add(mappedItem);
        }

        targetProperty.SetValue(result, mappedCollection);
    }

    private object? MapObject(Type targetType, object source, HashSet<object> visited)
    {
        var method = typeof(JMSMapper)
            .GetMethod(nameof(MapObjectGeneric), BindingFlags.Instance | BindingFlags.NonPublic)
            ?.MakeGenericMethod(targetType);

        return method?.Invoke(this, new object[] { source, visited });
    }

    private T? MapObjectGeneric<T>(object source, HashSet<object> visited) where T : class, new()
    {
        return MapObject<T>(source, visited);
    }

    private Type? GetCollectionItemType(Type collectionType)
    {
        if (collectionType.IsArray)
            return collectionType.GetElementType();

        if (collectionType.IsGenericType)
            return collectionType.GetGenericArguments().FirstOrDefault();

        return null;
    }

    private IList? CreateCollection(Type collectionType, Type itemType)
    {
        if (collectionType.IsArray)
            return Array.CreateInstance(itemType, 0) as IList;

        if (collectionType.IsGenericType)
            return Activator.CreateInstance(typeof(List<>).MakeGenericType(itemType)) as IList;

        return null;
    }

    private void MapComplexObject(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var mappedObject = MapObject(targetProperty.PropertyType, value, visited);
        targetProperty.SetValue(result, mappedObject);
    }

    private bool IsCollection(Type type)
    {
        return type != typeof(string) && typeof(IEnumerable).IsAssignableFrom(type);
    }

    private bool IsComplexType(Type type) //
    {
        return !type.IsPrimitive && !type.IsValueType && type != typeof(string);
    }

public interface IMappingExpression<TSource, TDestination> where TSource : class where TDestination : class, new()

public static class MapperExtensions
{
    public static IServiceCollection AddJMSMapper(this IServiceCollection services, Action<MapperConfiguration>? configure = null)
    {
        var config = new MapperConfiguration();
        configure?.Invoke(config);
        
        services.AddSingleton<IMapper>(provider => 
        {
            var logger = provider.GetService<Action<string, Exception>>();
            return new JMSMapper(config, logger);
        });
        
        return services;
    }

    public static void Initialize(Action<MapperConfiguration> configure)
    {
        var config = new MapperConfiguration();
        configure.Invoke(config);
        // Configuração é armazenada estáticamente no JMSMapper
        JMSMapper.SetConfiguration(config);
    }
}

public class MapperConfiguration
{
    internal Dictionary<(Type Source, Type Target), Dictionary<string, Func<object, object>>> CustomMappings { get; } = new();
    internal Dictionary<(Type Source, Type Target), Dictionary<string, string>> PropertyMappings { get; } = new();

    public IMappingExpression<TSource, TDestination> CreateMap<TSource, TDestination>() 
        where TSource : class 
        where TDestination : class, new()
    {
        return new MappingExpression<TSource, TDestination>(this);
    }
}

public class MappingExpression<TSource, TDestination> : IMappingExpression<TSource, TDestination> 
    where TSource : class 
    where TDestination : class, new()
{
    private readonly MapperConfiguration _config;

    public MappingExpression(MapperConfiguration config)
    {
        _config = config;
    }

    public IMappingExpression<TSource, TDestination> ForMember<TMember>(
        string destinationProperty,
        Func<TSource, TMember> mappingFunction)
    {
        var key = (typeof(TSource), typeof(TDestination));
        
        if (!_config.CustomMappings.TryGetValue(key, out var mappings))
        {
            mappings = new Dictionary<string, Func<object, object>>();
            _config.CustomMappings[key] = mappings;
        }
        
        mappings[destinationProperty] = src => mappingFunction((TSource)src)!;
        
        return this;
    }

    public IMappingExpression<TSource, TDestination> ForMember(
        string destinationProperty,
        string sourceProperty)
    {
        var key = (typeof(TSource), typeof(TDestination));
        
        if (!_config.PropertyMappings.TryGetValue(key, out var mappings))
        {
            mappings = new Dictionary<string, string>();
            _config.PropertyMappings[key] = mappings;
        }
        
        mappings[destinationProperty] = sourceProperty;
        
        return this;
    }
}

public class JMSMapper : MapperBase
{
    private static MapperConfiguration? _staticConfig;
    private readonly MapperConfiguration? _instanceConfig;
    private readonly Action<string, Exception>? _logger;

    public JMSMapper(MapperConfiguration? config = null, Action<string, Exception>? logger = null)
    {
        _instanceConfig = config;
        _logger = logger;
    }

    public static void SetConfiguration(MapperConfiguration config)
    {
        _staticConfig = config;
    }

    private MapperConfiguration? GetActiveConfig() => _instanceConfig ?? _staticConfig;

    protected override T? MapObject<T>(object source, HashSet<object> visited) where T : class, new()
    {
        if (source == null || visited.Contains(source)) return null;

        visited.Add(source);

        T result = new();

        var sourceType = source.GetType();
        var targetType = typeof(T);

        var sourceProperties = GetProperties(sourceType);
        var targetProperties = GetProperties(targetType);

        var config = GetActiveConfig();
        var customMappings = config?.CustomMappings;
        var propertyMappings = config?.PropertyMappings;

        foreach (var targetProperty in targetProperties)
        {
            if (!targetProperty.CanWrite) continue;

            try
            {
                // Verifica mapeamentos customizados primeiro
                var key = (sourceType, targetType);
                if (customMappings?.TryGetValue(key, out var mappings) == true && 
                    mappings.TryGetValue(targetProperty.Name, out var mappingFunc))
                {
                    var value = mappingFunc(source);
                    targetProperty.SetValue(result, ConvertValue(value, targetProperty.PropertyType));
                    continue;
                }

                // Verifica mapeamentos de propriedade
                var sourcePropertyName = GetMappedPropertyName(sourceType, targetType, targetProperty.Name, propertyMappings);
                var sourceProperty = sourceProperties.FirstOrDefault(p => 
                    string.Equals(p.Name, sourcePropertyName, StringComparison.OrdinalIgnoreCase));

                if (sourceProperty == null || !sourceProperty.CanRead) continue;

                var value = sourceProperty.GetValue(source);

                if (value == null) continue;

                if (IsCollection(targetProperty.PropertyType))
                {
                    MapCollection(targetProperty, result, value, visited);
                }
                else if (IsComplexType(targetProperty.PropertyType))
                {
                    MapComplexObject(targetProperty, result, value, visited);
                }
                else
                {
                    var convertedValue = ConvertValue(value, targetProperty.PropertyType);
                    targetProperty.SetValue(result, convertedValue);
                }
            }
            catch (Exception ex)
            {
                _logger?.Invoke($"Erro ao mapear propriedade '{targetProperty.Name}'", ex);
            }
        }

        return result;
    }

    private string GetMappedPropertyName(Type sourceType, Type targetType, string targetPropertyName, 
        Dictionary<(Type, Type), Dictionary<string, string>>? propertyMappings)
    {
        var key = (sourceType, targetType);
        if (propertyMappings?.TryGetValue(key, out var mappings) == true && 
            mappings.TryGetValue(targetPropertyName, out var sourcePropertyName))
        {
            return sourcePropertyName;
        }
        return targetPropertyName;
    }

    private void MapCollection(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var itemType = GetCollectionItemType(targetProperty.PropertyType);
        if (itemType == null || !(value is IEnumerable sourceCollection)) return;

        var mappedCollection = CreateCollection(targetProperty.PropertyType, itemType);
        if (mappedCollection == null) return;

        foreach (var item in sourceCollection)
        {
            if (item == null) continue;

            var mappedItem = MapObject(itemType, item, visited);
            if (mappedItem != null)
                mappedCollection.Add(mappedItem);
        }

        targetProperty.SetValue(result, mappedCollection);
    }

    private object? MapObject(Type targetType, object source, HashSet<object> visited)
    {
        var method = typeof(JMSMapper)
            .GetMethod(nameof(MapObjectGeneric), BindingFlags.Instance | BindingFlags.NonPublic)
            ?.MakeGenericMethod(targetType);

        return method?.Invoke(this, new object[] { source, visited });
    }

    private T? MapObjectGeneric<T>(object source, HashSet<object> visited) where T : class, new()
    {
        return MapObject<T>(source, visited);
    }

    private Type? GetCollectionItemType(Type collectionType)
    {
        if (collectionType.IsArray)
            return collectionType.GetElementType();

        if (collectionType.IsGenericType)
            return collectionType.GetGenericArguments().FirstOrDefault();

        return null;
    }

    private IList? CreateCollection(Type collectionType, Type itemType)
    {
        if (collectionType.IsArray)
            return Array.CreateInstance(itemType, 0) as IList;

        if (collectionType.IsGenericType)
        {
            var listType = typeof(List<>).MakeGenericType(itemType);
            return Activator.CreateInstance(listType) as IList;
        }

        return null;
    }

    private void MapComplexObject(PropertyInfo targetProperty, object result, object value, HashSet<object> visited)
    {
        var mappedObject = MapObject(targetProperty.PropertyType, value, visited);
        targetProperty.SetValue(result, mappedObject);
    }

    private bool IsCollection(Type type)
    {
        return type != typeof(string) && typeof(IEnumerable).IsAssignableFrom(type);
    }

    private bool IsComplexType(Type type)
    {
        return !type.IsPrimitive && !type.IsValueType && type != typeof(string);
    }
}

Problema no Mapeamento de Enums no JMSAutoMapper
O código atual tem uma limitação no tratamento de enums, especialmente quando o valor de origem não é um int. Vamos analisar e corrigir o problema.

Problema Identificado
A conversão de enum só funciona quando:

O valor de origem é exatamente um int

Não há tratamento para outros casos como:

String para enum

Outros tipos numéricos para enum

Enum para enum diferente

Solução Proposta
Vamos modificar o método ConvertValue para lidar com todos os cenários de enum:

csharp
protected object? ConvertValue(object? value, Type targetType)
{
    if (value == null)
    {
        if (Nullable.GetUnderlyingType(targetType) != null || !targetType.IsValueType)
            return null;
        return Activator.CreateInstance(targetType);
    }

    try
    {
        var underlyingType = Nullable.GetUnderlyingType(targetType) ?? targetType;

        // Se o tipo de destino for string, converte o valor para string
        if (underlyingType == typeof(string))
        {
            return value.ToString();
        }

        // Se o valor já for do tipo correto, retorna o valor
        if (underlyingType.IsInstanceOfType(value))
        {
            return value;
        }

        // Conversão para enums - tratamento completo
        if (underlyingType.IsEnum)
        {
            return ConvertToEnum(value, underlyingType);
        }

        // Conversão para outros tipos
        return Convert.ChangeType(value, underlyingType);
    }
    catch
    {
        return null;
    }
}

private object ConvertToEnum(object value, Type enumType)
{
    // Caso 1: Já é o enum correto (mas de tipo diferente)
    if (value.GetType().IsEnum)
    {
        return Enum.ToObject(enumType, (int)value);
    }

    // Caso 2: É string - faz parse
    if (value is string stringValue)
    {
        return Enum.Parse(enumType, stringValue, true);
    }

    // Caso 3: É número - converte
    if (value is int || value is short || value is byte || 
        value is long || value is uint || value is ushort || 
        value is sbyte || value is ulong)
    {
        return Enum.ToObject(enumType, value);
    }

    // Caso 4: É decimal/double/float - converte para int primeiro
    if (value is decimal || value is double || value is float)
    {
        return Enum.ToObject(enumType, Convert.ToInt32(value));
    }

    throw new InvalidOperationException($"Não foi possível converter {value.GetType().Name} para {enumType.Name}");
}
